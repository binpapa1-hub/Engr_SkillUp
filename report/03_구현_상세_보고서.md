## 4. JavaScript 기능

### 4.1 탭 전환 기능

#### 함수: `showTab(tabName)`
```javascript
function showTab(tabName) {
  // 모든 탭과 콘텐츠 숨기기
  const tabs = document.querySelectorAll('.nav-tab');
  const contents = document.querySelectorAll('.content');
  
  tabs.forEach(tab => tab.classList.remove('active'));
  contents.forEach(content => content.classList.remove('active'));
  
  // 선택한 탭과 콘텐츠 활성화
  event.target.classList.add('active');
  document.getElementById(tabName).classList.add('active');
  
  // 부서원 관리 탭일 경우 목록 새로고침
  if (tabName === 'members') {
    renderMemberList();
  }
}
```

### 4.2 부서원 관리 기능

#### 데이터 저장/로드
```javascript
const STORAGE_KEY = 'ecgf_members';

function getMembers() {
  const data = localStorage.getItem(STORAGE_KEY);
  return data ? JSON.parse(data) : [];
}

function saveMembers(members) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(members));
}
```

#### 부서원 등록
```javascript
function saveMember(event) {
  event.preventDefault();
  
  const members = getMembers();
  const editId = document.getElementById('editId').value;
  const memberData = {
    id: editId || Date.now().toString(),
    name: document.getElementById('memberName').value,
    primaryArchetype: document.getElementById('primaryArchetype').value,
    secondaryArchetype: document.getElementById('secondaryArchetype').value,
    years: parseInt(document.getElementById('years').value),
    level: document.getElementById('level').value
  };
  
  if (editId) {
    // 수정
    const index = members.findIndex(m => m.id === editId);
    if (index !== -1) {
      members[index] = memberData;
    }
  } else {
    // 추가
    members.push(memberData);
  }
  
  saveMembers(members);
  resetForm();
  renderMemberList();
}
```

#### 부서원 목록 렌더링
```javascript
function renderMemberList() {
  const members = getMembers();
  const listContainer = document.getElementById('memberList');
  
  if (members.length === 0) {
    listContainer.innerHTML = '<div class="empty-state">...</div>';
    return;
  }
  
  // 레벨별로 정렬
  const levelOrder = { 'L1': 1, 'L2': 2, 'L3': 3, 'L4': 4, 'L5': 5 };
  members.sort((a, b) => {
    const levelDiff = (levelOrder[b.level] || 0) - (levelOrder[a.level] || 0);
    if (levelDiff !== 0) return levelDiff;
    return b.years - a.years;
  });
  
  listContainer.innerHTML = members.map(member => `
    <div class="member-card">
      <!-- 카드 내용 -->
    </div>
  `).join('');
}
```

#### 부서원 수정
```javascript
function editMember(id) {
  const members = getMembers();
  const member = members.find(m => m.id === id);
  
  if (member) {
    document.getElementById('editId').value = member.id;
    document.getElementById('memberName').value = member.name;
    // ... 나머지 필드 채우기
  }
}
```

#### 부서원 삭제
```javascript
function deleteMember(id) {
  if (confirm('정말 삭제하시겠습니까?')) {
    const members = getMembers();
    const filtered = members.filter(m => m.id !== id);
    saveMembers(filtered);
    renderMemberList();
  }
}
```

## 5. 데이터 모델

### 5.1 부서원 데이터 구조
```typescript
interface Member {
  id: string;                    // 고유 ID (타임스탬프 기반)
  name: string;                  // 이름
  primaryArchetype: string;       // 주요 성향
  secondaryArchetype?: string;    // 보조 성향 (선택사항)
  years: number;                 // 근무 연차
  level: 'L1' | 'L2' | 'L3' | 'L4' | 'L5';  // 업무 레벨
}
```
